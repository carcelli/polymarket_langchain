### Database Schema

```sql
CREATE TABLE markets (
    id TEXT PRIMARY KEY,
    question TEXT,
    description TEXT,
    outcomes TEXT,          -- JSON array
    outcome_prices TEXT,    -- JSON array
    volume REAL,
    liquidity REAL,
    active BOOLEAN,
    end_date TEXT,
    last_updated TEXT
);

CREATE TABLE news (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    market_id TEXT,
    title TEXT,
    description TEXT,
    url TEXT,
    published_at TEXT,
    ingested_at TEXT,
    FOREIGN KEY(market_id) REFERENCES markets(id)
);
```
